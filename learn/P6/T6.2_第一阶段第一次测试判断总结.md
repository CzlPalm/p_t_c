下一步：ssl_st* 的 client_random 偏移探测
需要写一个探针，在 ssl_log_secret 触发时 dump args[0] 周围的内存，然后和 SSLKEYLOGFILE 里的 client_random 交叉比对。方法和 P3 修正 client_random 偏移时完全一样。
交给 Cursor 的探针代码：
javascript// probe_ssl_log_secret_cr.js
// 目标：找到 ssl_st* 上 client_random 的正确偏移
// 方法：dump args[0] 的多层内存，事后与 SSLKEYLOGFILE 比对

'use strict';
const mod = Process.getModuleByName('chrome');

let hitCount = 0;
const MAX_HITS = 10;

Interceptor.attach(mod.base.add(ptr('0x04883520')), {
    onEnter(args) {
        if (hitCount >= MAX_HITS) return;
        hitCount++;
        
        const ssl = args[0];
        let label = '?';
        try { label = args[1].readUtf8String(); } catch(_) {}
        
        // 只看前几次，每次 dump 三层内存
        console.log(`\n=== ssl_log_secret #${hitCount} label="${label}" ssl=${ssl} ===`);
        
        // 层1：ssl_st 直接内存（前 0x300 字节，每 0x10 一行）
        console.log('[L1] ssl_st 直接偏移 (寻找 32 字节 client_random):');
        try {
            const dump = ssl.readByteArray(0x300);
            const u8 = new Uint8Array(dump);
            for (let off = 0; off < 0x300; off += 0x10) {
                const slice = Array.from(u8.slice(off, off + 0x10))
                    .map(b => b.toString(16).padStart(2, '0')).join('');
                // 只打印非全零行
                if (slice !== '00000000000000000000000000000000') {
                    console.log(`  +0x${off.toString(16).padStart(3,'0')}: ${slice}`);
                }
            }
        } catch(e) { console.log(`  L1 读取失败: ${e}`); }
        
        // 层2：*(ssl + 偏移) 解引用后的内存
        const PTR_OFFSETS = [0x00, 0x08, 0x10, 0x18, 0x20, 0x28, 0x30, 0x38, 0x40, 0x48];
        for (const pOff of PTR_OFFSETS) {
            try {
                const p = ssl.add(pOff).readPointer();
                if (p.isNull() || p.compare(ptr('0x1000')) < 0) continue;
                const sub = p.readByteArray(0x100);
                const u8 = new Uint8Array(sub);
                // 只打印看起来像随机数据（高熵）的区域
                for (let off = 0; off < 0x100; off += 0x20) {
                    const slice = Array.from(u8.slice(off, off + 0x20))
                        .map(b => b.toString(16).padStart(2, '0')).join('');
                    const uniqueBytes = new Set(u8.slice(off, off + 0x20)).size;
                    if (uniqueBytes > 12) { // 高熵 = 可能是 client_random
                        console.log(`  [L2] *(ssl+0x${pOff.toString(16)})+0x${off.toString(16).padStart(3,'0')}: ${slice}`);
                    }
                }
            } catch(_) {}
        }
        
        // 层3：*(*(ssl+偏移)+偏移) 二级解引用
        for (const pOff of [0x00, 0x08, 0x10, 0x30]) {
            try {
                const p1 = ssl.add(pOff).readPointer();
                if (p1.isNull()) continue;
                for (const subOff of [0x00, 0x10, 0x20, 0x28, 0x30, 0x38, 0x40, 0x50, 0x60]) {
                    try {
                        const p2 = p1.add(subOff).readPointer();
                        if (p2.isNull() || p2.compare(ptr('0x1000')) < 0) continue;
                        const data = p2.readByteArray(0x40);
                        const u8 = new Uint8Array(data);
                        const uniqueBytes = new Set(u8.slice(0, 32)).size;
                        if (uniqueBytes > 16) {
                            const hex32 = Array.from(u8.slice(0, 32))
                                .map(b => b.toString(16).padStart(2, '0')).join('');
                            console.log(`  [L3] *(*(ssl+0x${pOff.toString(16)})+0x${subOff.toString(16)})+0x000: ${hex32}`);
                        }
                    } catch(_) {}
                }
            } catch(_) {}
        }
    }
});
